# Default example

This example deploys the module in its simplest form.

## Setup

To set up this module, review the architecture and guidance in the module documentation at (hello)[../.]

```hcl
# Basic usage of the Copilot Studio module. Assumes the usage of AVMs to initialize minimal prerequisite resources.
# See the CAIRA example for a more exhaustive resource initialization if needed.

locals {
  search_endpoint_url = "https://${azurerm_search_service.ai_search.name}.search.windows.net"
}

data "azurerm_client_config" "current" {}

# The unique ID that will be included in most resources managed by this module
resource "random_string" "name" {
  length  = 5
  numeric = false
  special = false
  upper   = false
}

# The Resource Group that will contain the resources managed by this module
resource "azurerm_resource_group" "this" {
  location = var.location
  name     = "rg-${random_string.name.id}"
}

module "copilot_studio" {
  source = "../.."

  # Basic properties
  name                = "Copilot + Azure AI"
  resource_group_name = azurerm_resource_group.this.name
  unique_id           = random_string.name.id # The unique ID to include with any resources generated by this module

  # Authentication inputs
  resource_share_user = var.resource_share_user
  principal_secret    = var.principal_secret

  # Network inputs
  primary_vnet_name    = module.primary_virtual_network.name
  primary_subnet_name  = var.primary_subnet_name
  failover_vnet_name   = module.failover_virtual_network.name
  failover_subnet_name = var.failover_subnet_name

  # Power Platform properties
  # Note: the Power Platform environment variables have default values that will 
  #  be used to generate new resources. If you have an existing environment you'd 
  #  prefer to manage instead, specify the environment IDs in the variable objects
  #  and the module will attempt to manage the existing environment instead. 
  power_platform_environment         = var.power_platform_environment
  power_platform_managed_environment = var.power_platform_managed_environment
  # Add Power Platform connections. For the MCS example we need the AOAI connection 
  # and AI Search connection, but this list could take any supported connectors.
  power_platform_connections = [
    {
      name         = "shared_azureopenai"
      display_name = "Azure OpenAI Connection"
      connection_parameters = {
        azureOpenAIResourceName = module.azure_open_ai.name
        azureOpenAIApiKey       = module.azure_open_ai.primary_access_key
        azureSearchEndpointUrl  = local.search_endpoint_url
        azureSearchApiKey       = azurerm_search_service.ai_search.primary_key
      }
      connector_name = "azureopenai"
    },
    {
      name         = "shared_azureaisearch"
      display_name = "Azure AI Search Connection"
      connection_parameters = {
        ConnectionEndpoint = local.search_endpoint_url
        AdminKey           = azurerm_search_service.ai_search.primary_key
      }
      connector_name = "azureaisearch"
    }
  ]
}

# Example tenant settings to support generative answers in Copilot.
# Given the sensitivity and different security of tenant settings,
# this resource is block commented but may be used if allowed and desired.

# resource "powerplatform_tenant_settings" "settings" {
#   power_platform = {
#     power_automate = {
#       disable_copilot = false
#       # Optional
#       disable_copilot_with_bing = false
#     }
#     intelligence = {
#       disable_copilot               = false
#       enable_open_ai_bot_publishing = true
#     }
#   }
# }
```

### <a name="input_app_insights_sections"></a> [app\_insights\_sections](#input\_app\_insights\_sections)

Description: A map of App Insights sections, each containing a KQL query, name, and chart type.

Type:

```hcl
map(object({
    query = string
    name  = string
    chart = string
  }))
```

Default:

```json
{
  "section_1": {
    "chart": "unstackedbar",
    "name": "Requests over time.",
    "query": "traces\n| where message contains \"Log Number\"\n| summarize requestCount = count() by bin(timestamp, 5s)\n| order by timestamp asc"
  },
  "section_2": {
    "chart": "barchart",
    "name": "All requests split into topics.",
    "query": "traces\n| where message contains \"Log Number\"\n| extend topic=customDimensions[\"topic\"]\n| summarize amount=count() by bin(timestamp, 10s), tostring(topic)"
  },
  "section_3": {
    "chart": "table",
    "name": "Last 20 requests.",
    "query": "traces\n| where message contains \"Log Number\"\n| extend topic=customDimensions[\"topic\"]\n| limit 20"
  },
  "section_4": {
    "chart": "timechart",
    "name": "Response time by timestamp.",
    "query": "traces\n| where message contains \"Log Number\"\n| extend app_timestamp_real = todouble(customDimensions[\"timestamp\"])\n| extend app_timestamp = unixtime_seconds_todatetime(app_timestamp_real)\n| extend time_difference = tolong(timestamp - app_timestamp)\n| summarize average=avg(time_difference) by bin(timestamp, 5s)"
  },
  "section_5": {
    "chart": "stat",
    "name": "Avg time per response overall.",
    "query": "traces\n| where message contains \"Log Number\"\n| extend app_timestamp_real = todouble(customDimensions[\"timestamp\"])\n| extend app_timestamp = unixtime_seconds_todatetime(app_timestamp_real)\n| extend time_difference = tolong(timestamp - app_timestamp)\n| summarize average=avg(time_difference)"
  },
  "section_6": {
    "chart": "barchart",
    "name": "Avg time per response per topic.",
```

### <a name="input_cognitive_deployments"></a> [cognitive\_deployments](#input\_cognitive\_deployments)

Description:   A map of cognitive model deployments to create on the Azure OpenAI Cognitive Services account. The map key is deliberately arbitrary to avoid issues where map keys maybe unknown at plan time.
  - `name` - (Required) The name of the deployment.
  - `model` - (Required) The model to deploy.
    - `format` - "OpenAI"  
    - `name` - The name of the model to deploy.
    - `version` - The version of the model to deploy.
  - `scale` - (Required) The scale of the model.
    - `type` - The type of scale to use. Possible values are `Standard`.
  - `rai_policy_name` - (Required) The name of the RAI policy to use for the deployment.  
  example:
  ```
  {
    "gpt-4" = {
      name = "gpt-4"
      model = {
        format  = "OpenAI"
        name    = "gpt-4"
        version = "0125-Preview"
      }
      scale = {
        type = "Standard"
      }
      rai_policy_name = "Microsoft.DefaultV2"
    }
    "text-embedding-ada-002" = {
      name = "text-embedding-ada-002"
      model = {
        format  = "OpenAI"
        name    = "text-embedding-ada-002"
        version = "2"
      }
      scale = {
        type = "Standard"
      }
      rai_policy_name = "Microsoft.DefaultV2"
    }
  }
```

Type:

```hcl
map(object({
    name = string
    model = object({
      format  = string
      name    = string
      version = string
    })
    scale = object({
      type = string
    })
    rai_policy_name = string
  }))
```

Default:

```json
{
  "gpt-4": {
    "model": {
      "format": "OpenAI",
      "name": "text-embedding-3-large",
      "version": "1"
    },
    "name": "text-embedding-3-large",
    "rai_policy_name": "Microsoft.DefaultV2",
    "scale": {
      "type": "Standard"
    }
  }
}
```
